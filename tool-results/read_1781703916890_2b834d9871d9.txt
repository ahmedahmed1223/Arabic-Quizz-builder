    19→// ─── Q1: PubSub Store ──────────────────────────────────────
    20→// Central state event bus. Lightweight (no Redux, no Proxy).
    21→// Usage:
    22→//   Store.subscribe('SCORE_CHANGE', (payload)=>{...});
    23→//   Store.dispatch('SCORE_CHANGE', {teamId, delta});
    24→// Backward-compatible: existing code that mutates `state` directly
    25→// still works; new code should dispatch through Store.
    26→const Store=(function(){
    27→  const listeners=new Map();  // eventName → [fn, ...]
    28→  const history=[];           // last 50 dispatched actions (debug)
    29→  const MAX_SUBSCRIBERS_PER_EVENT=50;  // V11: Prevent memory leaks from repeated subscriptions
    30→  return{
    31→    subscribe(event,fn){
    32→      if(!listeners.has(event))listeners.set(event,[]);
    33→      const arr=listeners.get(event);
    34→      // V11: Check for duplicate subscriptions (same event + same function)
    35→      if(arr.indexOf(fn)>=0)return ()=>{};  // Already subscribed, skip
    36→      // V11: Limit subscribers per event to prevent memory leaks
    37→      if(arr.length>=MAX_SUBSCRIBERS_PER_EVENT){
    38→        if(typeof ErrorBus!=='undefined')ErrorBus.capture(new Error('Max subscribers reached for: '+event),'Store.subscribe');
    39→        return ()=>{};
    40→      }
    41→      arr.push(fn);
    42→      return ()=>{  // unsubscribe
    43→        const a=listeners.get(event);
    44→        if(!a)return;
    45→        const i=a.indexOf(fn);
    46→        if(i>=0)a.splice(i,1);
    47→      };
    48→    },
    49→    dispatch(event,payload){
    50→      history.push({t:Date.now(),event,payload});
    51→      if(history.length>50)history.shift();
    52→      const arr=listeners.get(event);
    53→      if(!arr)return;
    54→      // V11: Wrap each subscriber call in try/catch so one failing subscriber doesn't break others
    55→      arr.forEach(fn=>{
    56→        try{fn(payload);}
    57→        catch(e){if(typeof ErrorBus!=='undefined')ErrorBus.capture(e,'Store:'+event);else console.error('[Store]',event,e);}
    58→      });
    59→    },
    60→    // V11: Unsubscribe all listeners for cleanup
    61→    unsubscribeAll(){
    62→      listeners.clear();
    63→    },
    64→    getHistory(){return[...history];},
    65→    _listenersCount(){return Array.from(listeners.entries()).map(([k,v])=>[k,v.length]);}

   269→// 6 channels: bgm (background music), sfx (correct/wrong), wheel (spin music),
   270→// question (audio question), tense (timer tension), podium (victory).
   271→// Rule: playing on a channel STOPS whatever was previously on that channel.
   272→const AudioMixer=(function(){
   273→  const channels={bgm:null,sfx:null,wheel:null,question:null,tense:null,podium:null};
   274→  const bgmState={wasPlayingBeforeDuck:false,duckedBy:null};

   276→  function _stopNode(node){
   277→    if(!node)return;
   278→    try{
   279→      if(typeof node.pause==='function'){node.pause();try{node.currentTime=0;}catch(e){try{ErrorBus.capture(e,"catch#AUTO_51")}catch(_){}}}
   280→      // V11: Release media resource to prevent memory leaks
   281→      try{node.src='';node.load();}catch(e){try{ErrorBus.capture(e,"catch#AUTO_52")}catch(_){}}
   282→      if(typeof node.stop==='function')node.stop();
   283→      if(typeof node.disconnect==='function')try{node.disconnect();}catch(e){try{ErrorBus.capture(e,"catch#AUTO_53")}catch(_){}}
   284→    }catch(e){ErrorBus.capture(e,'AudioMixer:_stopNode');}
   285→  }
   286→  
   287→  return{
   288→    // Play an audio src on a channel. Returns the created Audio element (or null).
   289→    // opts: {loop, volume (0-1), onEnd, src (required if node not provided), node}
   290→    play(channel,opts){
   291→      opts=opts||{};
   292→      if(!(channel in channels)){ErrorBus.capture(new Error('Unknown channel: '+channel),'AudioMixer.play');return null;}
   293→      // Stop previous
   294→      this.stop(channel);
   295→      let node;
   296→      if(opts.node){node=opts.node;}
   297→      else if(opts.src){
   298→        try{
   299→          node=new Audio(opts.src);
   300→          if(opts.loop)node.loop=true;
   301→          if(typeof opts.volume==='number')node.volume=Math.max(0,Math.min(1,opts.volume));
   302→          if(typeof opts.onEnd==='function')node.addEventListener('ended',opts.onEnd,{once:true});
   303→          // V11: Always release src on ended to prevent memory leaks
   304→          node.addEventListener('ended',function(){try{this.src='';this.load();}catch(e){try{ErrorBus.capture(e,"catch#AUTO_54")}catch(_){}}},{once:true});
   305→          node.play().catch(e=>{ErrorBus.capture(e,'AudioMixer.play:'+channel);});
   306→        }catch(e){ErrorBus.capture(e,'AudioMixer.play:ctor');return null;}
   307→      }
   308→      channels[channel]=node;
   309→      return node;
   310→    },
   311→    // Register an already-playing node on a channel (for legacy integration)
   312→    register(channel,node){
   313→      if(!(channel in channels))return false;
   314→      // V10-fix: skip stop() if the same node is already registered — prevents destroying the node being re-registered
   315→      if(channels[channel]!==node) this.stop(channel);
   316→      channels[channel]=node;
   317→      return true;
   318→    },
   319→    stop(channel){
   320→      if(!(channel in channels))return false;
   321→      if(channels[channel]){_stopNode(channels[channel]);channels[channel]=null;}
   322→      return true;
   323→    },
   324→    stopAll(){
   325→      Object.keys(channels).forEach(c=>this.stop(c));
   326→      bgmState.duckedBy=null;bgmState.wasPlayingBeforeDuck=false;
   327→    },
   328→    isPlaying(channel){return !!channels[channel];},
   329→    get(channel){return channels[channel]||null;},
   330→    // Duck the BGM to a lower volume while another channel plays; restore on unduck.
   331→    duck(byChannel,toVolume){
   332→      if(!channels.bgm)return false;
   333→      if(bgmState.duckedBy)return false;  // already ducked
   334→      bgmState.duckedBy=byChannel;
   335→      bgmState.prevVolume=channels.bgm.volume;
   336→      try{channels.bgm.volume=Math.max(0,Math.min(1,toVolume||0.2));}catch(e){try{ErrorBus.capture(e,"catch#AUTO_55")}catch(_){}}
   337→      return true;
   338→    },
   339→    unduck(byChannel){
   340→      if(bgmState.duckedBy!==byChannel)return false;
   341→      if(channels.bgm&&typeof bgmState.prevVolume==='number'){
   342→        try{channels.bgm.volume=bgmState.prevVolume;}catch(e){try{ErrorBus.capture(e,"catch#AUTO_56")}catch(_){}}
   343→      }
   344→      bgmState.duckedBy=null;
   345→      return true;
   346→    },
   347→    // Clean up only ended/paused channels without stopping active playback
   348→    cleanup(){
   349→      Object.keys(channels).forEach(c=>{
   350→        const n=channels[c];
   351→        if(n&&(n.ended||n.paused)){_stopNode(n);channels[c]=null;}
   352→      });
   353→    },
   354→    // State snapshot for diagnostics
   355→    snapshot(){
   356→      const out={};
   357→      Object.keys(channels).forEach(c=>{
   358→        const n=channels[c];
   359→        out[c]=n?{active:true,paused:n.paused===true,volume:n.volume,currentTime:n.currentTime}:null;
   360→      });
   361→      return out;
   362→    }
   363→  };

   493→    isLocked(key){return locked.has(key);},
